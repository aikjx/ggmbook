import numpy as np

# CODATA 2018 基本常数
c = 299792458 # 光速，m/s
epsilon0 = 8.8541878128e-12 # 真空介电常数，F/m
e = 1.602176634e-19 # 基本电荷，C
hbar = 1.054571817e-34 # 约化普朗克常数，J·s
G = 6.67430e-11 # 万有引力常数，m³·kg⁻¹·s⁻²
mp = 1.67262192369e-27 # 质子质量，kg

# 计算 Z'
Z_prime = c / (8 * np.pi * epsilon0)
print(f"Z' = &#123;Z_prime:.9e&#125; kg·m⁴·s⁻³·C⁻²")

# 计算 Z
Z = (G * c) / 2
print(f"Z = &#123;Z:.9e&#125; m⁴·kg⁻¹·s⁻³")

# 计算比值 Z'/Z
ratio = Z_prime / Z
print(f"Z'/Z = &#123;ratio:.9e&#125;")

# 计算精细结构常数 α
alpha_calc = (2 * e**2 * Z_prime) / (hbar * c**2)
alpha_exp = 7.2973525693e-3
print(f"α_calc = &#123;alpha_calc:.11e&#125;")
print(f"α_exp = &#123;alpha_exp:.11e&#125;")

# 计算相对误差
delta = abs(alpha_exp - alpha_calc) / alpha_exp * 100
print(f"相对误差 = &#123;delta:.6f&#125;%")

# 计算质子间电磁力与引力强度比
