# -*- coding: utf-8 -*-
"""
基于公式
 pi = lim_&#123;N,n->∞&#125; [ 2^N * (N/2)! / S(N, p_n) ]^(2/N)
的高精度数值验证程序

功能：
1. 高精度计算主公式
2. 支持 nth prime p_n
3. 支持自定义 S(N, p_n)
4. 支持 exact / Stirling 两种基准验证
5. 输出误差表
"""

import math
from typing import Callable, List, Dict, Any

import mpmath as mp

# =========================
# 1. 全局高精度设置
# =========================
mp.mp.dps = 80 # 小数精度，可调大

# =========================
# 2. 素数工具
# =========================
def is_prime(k: int) -> bool:
 """判断 k 是否为素数"""
 if k < 2:
 return False
 if k == 2:
 return True
 if k % 2 == 0:
 return False
 r = int(math.isqrt(k))
 for i in range(3, r + 1, 2):
 if k % i == 0:
 return False
 return True

def nth_prime(n: int) -> int:
 """返回第 n 个素数 p_n，规定 p_1 = 2"""
 if n < 1:
 raise ValueError("n 必须 >= 1")
 count = 0
 x = 1
 while count < n:
 x += 1
 if is_prime(x):
 count += 1
 return x

# =========================
# 3. Gamma / 阶乘工具
# =========================
def half_factorial(N: int) -> mp.mpf:
 """
 返回 (N/2)! = Gamma(N/2 + 1)
 注意：N 不一定必须偶数，Gamma 可自然延拓
 """
 return mp.gamma(mp.mpf(N) / 2 + 1)

def log_half_factorial(N: int) -> mp.mpf:
 """返回 log((N/2)!) = log Gamma(N/2 + 1)"""
 return mp.loggamma(mp.mpf(N) / 2 + 1)

# =========================
# 4. 主公式计算
# =========================
def pi_formula_value(N: int, n: int, S_func: Callable[[int, int], mp.mpf]) -> mp.mpf:
 """
 计算：
 [ 2^N * (N/2)! / S(N, p_n) ]^(2/N)
 """
 if N <= 0:
 raise ValueError("N 必须 > 0")
 if n <= 0:
 raise ValueError("n 必须 > 0")

 p_n = nth_prime(n)
 S_val = S_func(N, p_n)

 if S_val <= 0:
 raise ValueError(f"S(N, p_n) 必须为正数，当前 S=&#123;S_val&#125;")

 # 为避免溢出，走对数
 log_num = N * mp.log(2) + log_half_factorial(N)
 log_den = mp.log(S_val)
 log_expr = (mp.mpf(2) / N) * (log_num - log_den)
 return mp.e ** log_expr

def relative_error(x: mp.mpf, target: mp.mpf = mp.pi) -> mp.mpf:
 """相对误差"""
 return abs(x - target) / abs(target)

# =========================
# 5. 理论上“恰好成立”的 S(N,p_n)
# =========================
def S_exact_identity(N: int, p_n: int) -> mp.mpf:
 """
 若你希望该式对每个 N 都“恒等给出 pi”，
 那么反解得到：
 S(N, p_n) = 2^N * Gamma(N/2+1) / pi^(N/2)

 注意：
 这个 S 与 p_n 无关，因此它只是“构造型”验证，
 不是独立素数公式。
 """
 _ = p_n # 占位，表示当前版本未使用 p_n
 return mp.power(2, N) * mp.gamma(mp.mpf(N) / 2 + 1) / mp.power(mp.pi, mp.mpf(N) / 2)

# =========================
# 6. Stirling 渐近型 S(N,p_n)
# =========================
def S_stirling_asymptotic(N: int, p_n: int) -> mp.mpf:
 """
 由 Stirling 近似推出的渐近分母：
 S(N,p_n) ~ sqrt(pi*N) * (2N/(e*pi))^(N/2)

 同样这里不依赖 p_n，只是用于检验收敛趋势。
 """
 _ = p_n
 return mp.sqrt(mp.pi * N) * mp.power((2 * N) / (mp.e * mp.pi), mp.mpf(N) / 2)

# =========================
# 7. 一个示例：人为加入 p_n 的扰动项
# =========================
def S_with_prime_perturbation(N: int, p_n: int) -> mp.mpf:
 """
 示例函数：在 exact 基础上乘一个趋向 1 的素数扰动项
 S = S_exact * (1 + 1/(p_n*N))
 这样当 n,N 变大时，仍趋向原极限 pi。
 你可以把这里替换成你论文里的真实 S(N,p_n)。
 """
 base = S_exact_identity(N, p_n)
 correction = 1 + mp.mpf(1) / (p_n * N)
 return base * correction

# =========================
# 8. 批量验证
# =========================
def run_experiment(
 N_list: List[int],
 n_list: List[int],
 S_func: Callable[[int, int], mp.mpf],
 name: str = "custom"
) -> List[Dict[str, Any]]:
 """
 对一组 (N, n) 做数值试验
 """
 rows = []
 for N, n in zip(N_list, n_list):
 p_n = nth_prime(n)
 value = pi_formula_value(N, n, S_func)
 err = relative_error(value, mp.pi)

 rows.append(&#123;
 "name": name,
 "N": N,
 "n": n,
 "p_n": p_n,
 "value": value,
 "pi": mp.pi,
 "abs_error": abs(value - mp.pi),
 "rel_error": err,
 &#125;)
 return rows

def print_table(rows: List[Dict[str, Any]], digits: int = 18) -> None:
 """打印结果表"""
 table_width = 120
 sep = "=" * table_width
 print(sep)
 print(f"&#123;'name':<18&#125; &#123;'N':>4&#125; &#123;'n':>4&#125; &#123;'p_n':>6&#125; &#123;'value':>24&#125; &#123;'abs_error':>20&#125; &#123;'rel_error':>20&#125;")
 print(sep)
 for r in rows:
 print(
 f"&#123;r['name']:<18&#125; "
 f"&#123;r['N']:>4&#125; "
 f"&#123;r['n']:>4&#125; "
 f"&#123;r['p_n']:>6&#125; "
 f"&#123;mp.nstr(r['value'], digits):>24&#125; "
 f"&#123;mp.nstr(r['abs_error'], digits):>20&#125; "
 f"&#123;mp.nstr(r['rel_error'], digits):>20&#125;"
 )
 print(sep)

# =========================
# 9. 收敛阶简单观察
# =========================
def convergence_report(
 N_values: List[int],
 n_rule: Callable[[int], int],
 S_func: Callable[[int, int], mp.mpf],
 label: str
) -> None:
 table_width = 110
 print(f"\n&#123;'=' * table_width&#125;")
 print(f" 收敛报告: &#123;label&#125;")
 print(f"&#123;'=' * table_width&#125;")
 print(f" &#123;'N':>5&#125; &#123;'n':>5&#125; &#123;'value':>28&#125; &#123;'rel_error':>18&#125; &#123;'err_ratio':>14&#125;")
 print(f"&#123;'=' * table_width&#125;")
 last_err = None
 for N in N_values:
 n = n_rule(N)
 val = pi_formula_value(N, n, S_func)
 err = relative_error(val)
 if last_err is None:
 ratio_text = "N/A"
 else:
 ratio = err / last_err if last_err != 0 else mp.nan
 ratio_text = mp.nstr(ratio, 10)
 print(
 f" &#123;N:>5&#125; &#123;n:>5&#125; &#123;mp.nstr(val, 20):>28&#125; &#123;mp.nstr(err, 12):>18&#125; &#123;ratio_text:>14&#125;"
 )
 last_err = err
 print(f"&#123;'=' * table_width&#125;\n")

# =========================
# 10. 用户替换区
# =========================
def S_user_defined(N: int, p_n: int) -> mp.mpf:
 """
 这里留给你替换成论文中的真实 S(N,p_n)。

 你只要把 return 后面改成你的公式即可。

 示例：
 return some_expression_in_N_and_pn

 当前默认先用“素数微扰型”演示。
 """
 return S_with_prime_perturbation(N, p_n)

# =========================
# 11. 主程序
# =========================
def main():
 print("高精度位数 dps =", mp.mp.dps)
 print("目标常数 pi =", mp.nstr(mp.pi, 30))

 # 测试点
 N_list = [10, 20, 50, 100, 200, 500, 1000]
 n_list = [10, 20, 50, 100, 200, 500, 1000]

 # 1) 恒等反解型 S
 rows_exact = run_experiment(N_list, n_list, S_exact_identity, "S_exact_identity")
 print_table(rows_exact)

 # 2) Stirling 渐近型 S
 rows_stirling = run_experiment(N_list, n_list, S_stirling_asymptotic, "S_stirling")
 print_table(rows_stirling)

 # 3) 含素数扰动的示例 S
 rows_prime = run_experiment(N_list, n_list, S_with_prime_perturbation, "S_prime_perturb")
 print_table(rows_prime)

 # 4) 你自己的 S
 rows_user = run_experiment(N_list, n_list, S_user_defined, "S_user_defined")
 print_table(rows_user)

 # 收敛观察
 convergence_report(
 N_values=[10, 20, 50, 100, 200, 500, 1000],
 n_rule=lambda N: N,
 S_func=S_stirling_asymptotic,
 label="Stirling asymptotic S"
 )

 convergence_report(
 N_values=[10, 20, 50, 100, 200, 500, 1000],
 n_rule=lambda N: N,
 S_func=S_with_prime_perturbation,
 label="Prime perturbation S"
 )

if __name__ == "__main__":
 main()
