### 附录：准化时间势差精算Python实现

import math

# ====================== 物理常量定义 ======================
G = 6.674e-11 # 万有引力常量 (N·m²/kg²)
c = 299792458 # 真空中的光速 (m/s)
M_EARTH = 5.972e24 # 地球质量 (kg)
R_EARTH = 6.371e6 # 地球赤道半径 (m)
DAY_SECONDS = 86400 # 单日秒数 (s)

# ====================== 核心计算函数 ======================
def time_potential_exact(v: float, r: float, M: float = M_EARTH) -> float:
 """
 全维统一时间势精确计算
 :param v: 物体三维运动速度 (m/s)
 :param r: 物体到中心天体质心的径向距离 (m)
 :param M: 中心天体质量 (kg)，默认地球质量
 :return: 时间势 dτ/dt
 """
 term_v = (v ** 2) / (c ** 2)
 term_phi = (2 * G * M) / (r * c ** 2)
 return math.sqrt(1 - term_v - term_phi)

def time_potential_approx(v: float, r: float, M: float = M_EARTH) -> float:
 """
 低速弱场一阶近似时间势计算
 :param v: 物体三维运动速度 (m/s)
 :param r: 物体到中心天体质心的径向距离 (m)
 :param M: 中心天体质量 (kg)，默认地球质量
 :return: 近似时间势 dτ/dt
 """
 term_v = (v ** 2) / (2 * c ** 2)
 term_phi = (G * M) / (r * c ** 2)
 return 1 - term_phi - term_v

def time_difference_exact(obj1: tuple, obj2: tuple, M: float = M_EARTH) -> tuple:
 """
 两物体全维统一时间势差精确计算
 :param obj1: 物体1参数 (v1, r1)
 :param obj2: 物体2参数 (v2, r2)
 :param M: 中心天体质量 (kg)，默认地球质量
 :return: (单位坐标时时间差, 单日累计时间差(微秒))
 """
 v1, r1 = obj1
 v2, r2 = obj2
 tau1 = time_potential_exact(v1, r1, M)
 tau2 = time_potential_exact(v2, r2, M)
 delta_per_second = tau1 - tau2
 delta_day_us = delta_per_second * DAY_SECONDS * 1e6
 return delta_per_second, delta_day_us
